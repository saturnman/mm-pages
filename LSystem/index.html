from math import sin, cos, pi
import matplotlib.pyplot as pl
from matplotlib import collections
import numpy
class L_System(object):
    def __init__(self, rule):
        info = rule['S']
        for i in range(rule['iter']):
            ninfo = []
            for c in info:
                if c in rule:
                    ninfo.append(rule[c])
                else:
                    ninfo.append(c)
            info = "".join(ninfo)
        self.rule = rule
        self.info = info
    def get_lines(self):
        d = self.rule['direct']
        a = self.rule['angle']
        p = (0.0, 0.0)
        l = 1.0
        lines = []
        stack = []
        for c in self.info:
            if c in "Ff":
                r = d * pi / 180
                t = p[0] + l*cos(r), p[1] + l*sin(r)
                lines.append(((p[0], p[1]), (t[0], t[1])))
                p = t
            elif c == "+":
                d += a
            elif c == "-":
                d -= a
            elif c == "[":
                stack.append((p,d))
            elif c == "]":
                p, d = stack[-1]
                del stack[-1]
        return lines
    @staticmethod
    def get_Barnsley_lines():
        lines = []
        p = (0.0, 0.0)
        l = 1.0
        for i in range(500000):
            rnd = numpy.random.random()
            if rnd < 0.01:
                t = 0,0.16*p[1]
            elif rnd < 0.86:
                t = 0.85*p[0]+0.04*p[1],-0.04*p[0]+0.85*p[1]+1.6
            elif rnd < 0.93:
                t = 0.2*p[0]-0.26*p[1],0.23*p[0]+0.22*p[1]+1.6
            else:
                t = -0.15*p[0]+0.28*p[1],0.26*p[0]+0.24*p[1]+0.44
            lines.append((t[0], t[1]))
            p = t
        return lines

def draw(ax, rule, iter=None):
    if iter!=None:
        rule["iter"] = iter
    lines = L_System(rule).get_lines()
    linecollections = collections.LineCollection(lines)
    ax.add_collection(linecollections, autolim=True)
    ax.axis("equal")
    ax.set_axis_off()
    ax.set_xlim(ax.dataLim.xmin, ax.dataLim.xmax)
    ax.invert_yaxis()

import matplotlib
import matplotlib.pyplot as plt
import numpy as nm
fig = plt.gcf()
#fig.set_size_inches(150,75)
ITERATIONS = 30
#SIZE = lena.shape[0]
#:
SIZE = 500
MAX_COLOR = 255.
x_min, x_max = -2.5, 1
y_min, y_max = -1.2, 1.2
x,y = nm.meshgrid(nm.linspace(x_min,x_max,2*SIZE),nm.linspace(y_min,y_max,SIZE));
c = x + 1j*y
z = c.copy()
fractal = nm.zeros(z.shape,dtype=nm.uint8)+MAX_COLOR
for n in range(ITERATIONS):
    mask = nm.abs(z) <=10
    z[mask] = z[mask]**2+c[mask]
    fractal[(fractal==MAX_COLOR) & (~mask)] = (MAX_COLOR-1)*n/ITERATIONS
# Display the fractal
DPI = fig.get_dpi()
fig.set_size_inches(1200.0 / float(DPI), 800.0 / float(DPI))
im = plt.imshow(fractal,cmap=plt.get_cmap('flag'))
plt.title('Mandelbrot')
plt.axis('off')
